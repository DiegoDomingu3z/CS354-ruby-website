<?php
function fibonacci($n) {
	if ($n == 0)
		return 0;
	else if ($n == 1)
		return 1;
	else
		return fibonacci($n-1) + fibonacci($n-2);
}
for($i = 0; $i<20; $i++) {
	$value = fibonacci($i);
	echo "{$value}<br/>";
}

Example 5: 

        
<?php
function fetchRandomDogBreed()
{
    $url = "https://dog.ceo/api/breeds/image/random";

    $json = @file_get_contents($url);
	if ($json === false) {
	    die("Failed to fetch dog image.");
	}
	
	$data = json_decode($json, true);
	
	if (!is_array($data) || empty($data["message"] || empty($data["status"] || $data["status"] != "success"))) {
	    die("Invalid API response.");
	}
	
	return $data["message"];
}
$dogImage = fetchRandomDogBreed();
?>
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Random Dog Breed</title>
</head>
<body>

<?php if (!empty($dogImage)): ?>
    <img src="<?php echo htmlspecialchars($dogImage, ENT_QUOTES, 'UTF-8'); ?>" alt="Random dog" height="500px">
<?php else: ?>
    <p>No dog breed available right now.</p>
<?php endif; ?>
</body>
</html>
