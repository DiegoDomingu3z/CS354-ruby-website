<?php
function fibonacci($n) {
	if ($n == 0)
		return 0;
	else if ($n == 1)
		return 1;
	else
		return fibonacci($n-1) + fibonacci($n-2);
}
for($i = 0; $i<20; $i++) {
	$value = fibonacci($i);
	echo "{$value}<br/>";

<?php
function fetchRandomMeme(): ?array
{
    $url = 'https://meme-api.com/gimme';

    $json = @file_get_contents($url);
    if ($json === false) {
        return null;
    }

    $data = json_decode($json, true);
    if (!is_array($data)) {
        return null;
    }

    return $data;
}

$max   = 5;
$meme       = null;
$isSafeMeme = false;

for ($i = 0; $i < $max; $i++) {
    $meme = fetchRandomMeme();

    if ($meme === null) {
        continue;
    }

    $isSafeMeme = empty($meme['nsfw']) || $meme['nsfw'] === false;

    if ($isSafeMeme && !empty($meme['url'])) {
        break;
    }
}
?>
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Random Meme</title>
</head>
<body>

<?php if ($isSafeMeme && !empty($meme['url'])): ?>
    <img src="<?php echo htmlspecialchars($meme['url'], ENT_QUOTES, 'UTF-8'); ?>" alt="Random meme">
<?php else: ?>
    <p>No safe meme available right now.</p>
<?php endif; ?>
</body>
</html>
